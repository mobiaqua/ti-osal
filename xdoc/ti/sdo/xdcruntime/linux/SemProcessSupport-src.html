    42    import xdc.runtime.Assert;
    43    import xdc.runtime.Diags;
    44    import xdc.runtime.Error;
    45    import xdc.runtime.Log;
    46    import xdc.runtime.knl.ISemProcessSupport;
    47    
    48    @ModuleStartup          /* initialize statically allocated instances */
    49    @InstanceInitError      /* because initialization can fail */
    50    @InstanceFinalize       /* have to clean up on delete */
    51    module SemProcessSupport inherits ISemProcessSupport
    52    {
    53    
    54        /*! Asserted in SemProcess_pend */
    55        config Assert.Id A_unsupportedTimeout = {
    56            msg: "A_unsupportedTimeout: Only SemProcess_FOREVER is supported"
    57        };
    58    
    59        /*! Asserted in SemProcess_create */
    60        config Assert.Id A_invalidKey = {
    61            msg: "A_invalidKey: params key has not been set"
    62        };
    63    
    64        config Assert.Id A_POSIX_Error = {
    65            msg: "A_POSIX_Error: a POSIX function returned failure"
    66        };
    67    
    68        config Error.Id E_POSIX_Error = {
    69            msg: "E_POSIX_Error: a POSIX function returned failure, errno is %d"
    70        };
    71    
    72        config Log.Event L_extant = {
    73            mask: Diags.LIFECYCLE,
    74            msg: "--- SemProcess(%d) already exists, incrementing reference count"
    75        };
    76    
    77        config Log.Event L_create = {
    78            mask: Diags.LIFECYCLE,
    79            msg: "--- SemProcess(%d) DNE, creating"
    80        };
    81    
    82        config Int perms = 0666; /*! unix-style RW perms for the underlying sem */
    83    
    84    internal:
    85    
    86        const Int POSIX_FAILURE = 1;
    87    
    88        struct Instance_State {
    89            Int id;     /*! ID of SysV semaphore */
    90            Int key;    /*! Key of SysV semaphore */
    91            Int count;  /*! used only for data transport from red to blue code */
    92            Mode mode;
    93        };
    94    }
